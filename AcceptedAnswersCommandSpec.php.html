<?php

namespace spec\App\Application\Answers;

use App\Application\Command;
use App\Application\Questions\ChangeQuestionCommand;
use App\Domain\Questions\Question\QuestionId;
use PhpSpec\ObjectBehavior;

class AcceptedAnswersCommandSpec extends ObjectBehavior
{
    public $title;
    public $body;

    function let()
    {
        $this->title = 'title';
        $this->body = 'body';
        $this->beConstructedWith($this->title, $this->body);
    }

    function it_is_initializable()
    {
        $this->shouldHaveType(AcceptedAnswersCommand::class);
    }

    function its_a_command()
    {
        $this->shouldBeAnInstanceOf(Command::class);
    }

    function it_has_a_title()
    {
        $this->title()->shouldBe($this->title);
    }

    function it_has_a_body()
    {
        $this->body()->shouldBe($this->body);
    }
}
